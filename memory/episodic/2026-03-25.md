### 核心算法

**磁场计算（偶极子磁场）**：
```javascript
function calculateMagneticField(x, y) {
    let fx = 0, fy = 0;
    
    magneticPoles.forEach(pole => {
        const dx = pole.x - x;
        const dy = pole.y - y;
        const dist = Math.sqrt(dx * dx + dy * dy);
        
        if (dist > 10) {
            const force = pole.strength / (dist * dist);
            const angle = Math.atan2(dy, dx);
            
            if (pole.type === 'positive') {
                fx += Math.cos(angle) * force;
                fy += Math.sin(angle) * force;
            } else {
                fx -= Math.cos(angle) * force;
                fy -= Math.sin(angle) * force;
            }
        }
    });
    
    return { fx, fy };
}
```

**粒子运动**：

**PQRST波形生成**：
```javascript
function generatePQRST(phase) {
    const t = phase % 1;
    
    // P波（小波峰）
    if (t < 0.1) {
        return Math.sin(t * Math.PI * 10) * 0.15;
    }
    // PR段（平直）
    else if (t < 0.15) {
        return 0;
    }
    // Q波（小下降）
    else if (t < 0.18) {
        return -(t - 0.15) * 3;
    }
    // R波（大尖峰）
    else if (t < 0.22) {
        return Math.sin((t - 0.18) * Math.PI * 12.5) * 1;
    }
    // S波（小下降）
    else if (t < 0.25) {
        return -Math.sin((t - 0.22) * Math.PI * 16.7) * 0.2;
    }
    // ST段（平直）
    else if (t < 0.35) {
        return 0;
    }
    // T波（中等波峰）
    else if (t < 0.5) {
        return Math.sin((t - 0.35) * Math.PI * 6.67) * 0.3;
    }
    // 基线
    else {
        return 0;
    }
}
```

### 核心算法

**Navier-Stokes方程简化版**：
- **扩散（Diffusion）**：流体的扩散效应
- **平流（Advection）**：流体随速度场移动
- **投影（Projection）**：保持流体不可压缩性

**扩散**：

```javascript
function diffuse(b, x, x0, diff) {
    const a = DT * diff * (gridWidth - 2) * (gridHeight - 2);
    linearSolve(b, x, x0, a, 1 + 6 * a);
}
```

**平流**：
```javascript
function advect(b, d, d0, velocX, velocY) {
    // 根据速度场追溯粒子来源
    const dtx = DT * (gridWidth - 2);
    const dty = DT * (gridHeight - 2);
    
    // 双线性插值
    d[IX(i, j)] = s0 * (t0 * d0[IX(i0i, j0i)] + t1 * d0[IX(i0i, j1i)]) +
                  s1 * (t0 * d0[IX(i1i, j0i)] + t1 * d0[IX(i1i, j1i)]);
}
```

**投影**：
```javascript
function project(velocX, velocY, p, div) {
    // 计算散度
    div[IX(i, j)] = -0.5 * (
        velocX[IX(i + 1, j)] - velocX[IX(i - 1, j)] +
        velocY[IX(i, j + 1)] - velocY[IX(i, j - 1)]
    ) / gridWidth;
    
    // 求解压力场
    linearSolve(0, p, div, 1, 6);
    
    // 从速度场中减去压力梯度
    velocX[IX(i, j)] -= 0.5 * (p[IX(i + 1, j)] - p[IX(i - 1, j)]) * gridWidth;
    velocY[IX(i, j)] -= 0.5 * (p[IX(i, j + 1)] - p[IX(i, j - 1)]) * gridHeight;
}
```
